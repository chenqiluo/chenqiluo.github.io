  function drawKline(){
    var canvas = document.getElementById('sr-kline');
    if(!canvas) return;
    var isDark = document.documentElement.classList.contains('dark');
    var W = canvas.parentElement.clientWidth || 800;
    var H = 300;
    canvas.width = W;
    canvas.height = H;
    var ctx = canvas.getContext('2d');

    var pad = {l:52, r:16, t:20, b:40};
    var cw = W - pad.l - pad.r;
    var ch = H - pad.t - pad.b;

    ctx.fillStyle = isDark ? '#1e2129' : '#fff';
    ctx.fillRect(0,0,W,H);

    // 显示最近60根
    var show = kdata.slice(-60);
    var n = show.length;
    var allPrices = [];
    show.forEach(function(k){ allPrices.push(k.h, k.l); });
    var pMin = Math.min.apply(null, allPrices);
    var pMax = Math.max.apply(null, allPrices);
    var pRange = pMax - pMin || 1;
    pMin = pMin - pRange * 0.05;
    pMax = pMax + pRange * 0.05;
    pRange = pMax - pMin;

    function toY(p){ return pad.t + ch * (1 - (p - pMin) / pRange); }
    function toX(i){ return pad.l + (i + 0.5) * (cw / n); }

    // 网格线
    var gridColor = isDark ? '#2d3139' : '#f0f2f5';
    var textColor = isDark ? '#9aa0a6' : '#5f6368';
    ctx.strokeStyle = gridColor;
    ctx.lineWidth = 1;
    for(var gi=0; gi<=4; gi++){
      var yy = pad.t + ch * gi / 4;
      ctx.beginPath(); ctx.moveTo(pad.l, yy); ctx.lineTo(pad.l+cw, yy); ctx.stroke();
      var pv = pMax - pRange * gi / 4;
      ctx.fillStyle = textColor;
      ctx.font = '10px sans-serif';
      ctx.textAlign = 'right';
      ctx.fillText(pv.toFixed(1), pad.l - 4, yy + 3);
    }

    // MA5 MA20 线
    var colors5 = '#ff9800', colors20 = '#1a73e8';
    function drawMA(period, color){
      ctx.beginPath();
      ctx.strokeStyle = color;
      ctx.lineWidth = 1;
      var started = false;
      for(var i=0; i<n; i++){
        if(i < period-1){ continue; }
        var sum = 0;
        for(var j=0; j<period; j++) sum += show[i-j].c;
        var ma = sum / period;
        if(!started){ ctx.moveTo(toX(i), toY(ma)); started=true; }
        else ctx.lineTo(toX(i), toY(ma));
      }
      ctx.stroke();
    }
    drawMA(5, colors5);
    drawMA(20, colors20);

    // K线柱
    var bw = Math.max(2, cw/n * 0.6);
    for(var i=0; i<n; i++){
      var k = show[i];
      var x = toX(i);
      var isUp = k.c >= k.o;
      ctx.strokeStyle = isUp ? '#34a853' : '#ea4335';
      ctx.fillStyle   = isUp ? '#34a853' : '#ea4335';
      // 上下影线
      ctx.lineWidth = 1;
      ctx.beginPath();
      ctx.moveTo(x, toY(k.h));
      ctx.lineTo(x, toY(k.l));
      ctx.stroke();
      // 实体
      var yOpen  = toY(k.o);
      var yClose = toY(k.c);
      var rH = Math.abs(yClose - yOpen) || 1;
      ctx.fillRect(x - bw/2, Math.min(yOpen, yClose), bw, rH);
    }

    // X轴标签（每10根标一次）
    ctx.fillStyle = textColor;
    ctx.font = '10px sans-serif';
    ctx.textAlign = 'center';
    for(var i=0; i<n; i++){
      if(i % 10 === 0 || i === n-1){
        ctx.fillText(show[i].d, toX(i), H - pad.b + 14);
      }
    }

    // 图例
    ctx.font = '11px sans-serif'; ctx.textAlign='left';
    ctx.fillStyle = colors5;
    ctx.fillRect(pad.l+4, 6, 18, 2);
    ctx.fillText('MA5', pad.l+26, 12);
    ctx.fillStyle = colors20;
    ctx.fillRect(pad.l+54, 6, 18, 2);
    ctx.fillText('MA20', pad.l+76, 12);
  }

  if(document.readyState === 'loading'){
    document.addEventListener('DOMContentLoaded', drawKline);
  } else {
    drawKline();
  }
  window.addEventListener('resize', drawKline);
})();
