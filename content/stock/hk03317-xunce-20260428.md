function calcMA(arr, period){
var res = [];
for(var i=0;i<arr.length;i++){
if(i<period-1){res.push(null);continue;}
var sum=0; for(var j=i-period+1;j<=i;j++) sum+=arr[j];
res.push(sum/period);
}
return res;
}
